001package lombok.maven;
002
003import java.io.File;
004import java.io.IOException;
005import java.lang.reflect.InvocationTargetException;
006import java.nio.charset.Charset;
007import java.nio.charset.UnsupportedCharsetException;
008import java.util.ArrayList;
009import java.util.List;
010import java.util.Map;
011
012import org.apache.commons.lang3.StringUtils;
013import org.apache.maven.artifact.Artifact;
014import org.apache.maven.plugin.AbstractMojo;
015import org.apache.maven.plugin.MojoExecutionException;
016import org.apache.maven.plugin.logging.Log;
017import org.apache.maven.plugins.annotations.Component;
018import org.apache.maven.plugins.annotations.Parameter;
019import org.apache.maven.project.MavenProject;
020import org.sonatype.plexus.build.incremental.BuildContext;
021
022import lombok.launch.Delombok;
023
024/**
025 * Abstract mojo to Delombok java source with lombok annotations.
026 *
027 * @author <a href="mailto:anthony@whitford.com">Anthony Whitford</a>
028 * @see <a href="http://projectlombok.org/features/delombok.html">Delombok</a>
029 */
030public abstract class AbstractDelombokMojo extends AbstractMojo {
031
032    /**
033     * Specifies whether the delombok generation should be skipped.
034     */
035    @Parameter(property="lombok.delombok.skip", defaultValue="false", required=true)
036    protected boolean skip;
037
038    /**
039     * Encoding.
040     */
041    @Parameter(property="lombok.encoding", defaultValue="${project.build.sourceEncoding}", required=true)
042    protected String encoding;
043
044    /**
045     * Verbose flag.  Print the name of each file as it is being delombok-ed.
046     */
047    @Parameter(property="lombok.verbose", defaultValue="false", required=true)
048    protected boolean verbose;
049 
050    /**
051     * Add output directory flag.  Adds the output directory to the Maven build path.
052     */
053    @Parameter(property="lombok.addOutputDirectory", defaultValue="true", required=true)
054    protected boolean addOutputDirectory;
055
056    /**
057     * Formatting preferences.
058     */
059    @Parameter
060    protected Map<String, String> formatPreferences;
061
062    /**
063     * The Maven project to act upon.
064     */
065    @Parameter(property="project", required=true, readonly=true)
066    protected MavenProject project;
067
068    /**
069     * The plugin dependencies.
070     */
071    @Parameter(property="plugin.artifacts", required=true, readonly=true)
072    private List<Artifact> pluginArtifacts;
073
074    /**
075     * Build Context for improved Maven-Eclipse integration.
076     */
077    @Component
078    private BuildContext buildContext;
079
080    protected abstract String getGoalDescription ();
081
082    protected abstract File getOutputDirectory();
083
084    protected abstract File getSourceDirectory();
085
086    protected abstract String getSourcePath();
087
088    protected abstract void addSourceRoot(String path);
089
090    @Override
091    public void execute() throws MojoExecutionException {
092        final Log logger = getLog();
093        assert null != logger;
094
095        final String goal = getGoalDescription();
096        logger.debug("Starting " + goal);
097        final File outputDirectory = getOutputDirectory();
098        logger.debug("outputDirectory: " + outputDirectory);
099        final File sourceDirectory = getSourceDirectory();
100        logger.debug("sourceDirectory: " + sourceDirectory);
101        final String sourcePath = getSourcePath();
102        logger.debug("sourcePath: " + sourcePath);
103
104        if (this.skip) {
105            logger.warn("Skipping " + goal);
106        } else if (sourceDirectory.exists()) {
107            // Build a classPath for delombok...
108            final StringBuilder classPathBuilder = new StringBuilder();
109            for (final Object artifact : project.getArtifacts()) {
110                classPathBuilder.append(((Artifact)artifact).getFile()).append(File.pathSeparatorChar);
111            }
112            for (final Artifact artifact : pluginArtifacts) {
113                classPathBuilder.append(artifact.getFile()).append(File.pathSeparatorChar);
114            }
115            final String classPath = classPathBuilder.toString();
116            logger.debug("classpath: " + classPath);
117            try {
118                final Delombok delombok = new Delombok();
119                delombok.setVerbose(this.verbose);
120                delombok.setClasspath(classPath);
121
122                if (StringUtils.isNotBlank(this.encoding)) {
123                    try {
124                        delombok.setCharset(this.encoding);
125                    } catch (final UnsupportedCharsetException e) {
126                        logger.error("The encoding parameter is invalid; Please check!", e);
127                        throw new MojoExecutionException("Unknown charset: " + this.encoding, e);
128                    }
129                } else {
130                    logger.warn("No encoding specified; using default: " + Charset.defaultCharset());
131                }
132
133                if (null != formatPreferences && !formatPreferences.isEmpty()) {
134                    try {
135                        // Construct a list array just like the command-line option...
136                        final List<String> formatOptions = new ArrayList<String>(formatPreferences.size());
137                        for (final Map.Entry<String, String> entry : formatPreferences.entrySet()) {
138                            final String key = entry.getKey();
139                            // "pretty" is an exception -- it has no value...
140                            formatOptions.add( "pretty".equalsIgnoreCase(key) ? key : (key + ':' + entry.getValue()) );
141                        }
142                        delombok.setFormatPreferences(delombok.formatOptionsToMap(formatOptions));
143                    } catch (final Exception e) {
144                        logger.error("The formatPreferences parameter is invalid; Please check!", e);
145                        throw new MojoExecutionException("Invalid formatPreferences: " + this.formatPreferences, e);
146                    }
147                }
148
149                try {
150                    delombok.setOutput(outputDirectory);
151                    delombok.setSourcepath(getSourcePath());
152                    delombok.addDirectory(sourceDirectory);
153                    if (buildContext.hasDelta(sourceDirectory)) {
154                        delombok.delombok();
155                        logger.info(goal + " complete.");
156                    
157                        if (this.addOutputDirectory) {
158                            // adding generated sources to Maven project
159                            addSourceRoot(outputDirectory.getCanonicalPath());
160                            // Notify build context about a file created, updated or deleted...
161                            buildContext.refresh(outputDirectory);
162                        }
163                    } else {
164                        logger.info(goal + " skipped; No deltas detected.");
165                    }
166                } catch (final IOException e) {
167                    logger.error("Unable to delombok!", e);
168                    throw new MojoExecutionException("I/O problem during delombok", e);
169                }
170            } catch (final ClassNotFoundException e) {
171                throw new MojoExecutionException("Unable to delombok", e);
172            } catch (final IllegalAccessException e) {
173                throw new MojoExecutionException("Unable to delombok", e);
174            } catch (final InvocationTargetException e) {
175                throw new MojoExecutionException("Unable to delombok", e);
176            } catch (final InstantiationException e) {
177                throw new MojoExecutionException("Unable to delombok", e);
178            } catch (final NoSuchMethodException e) {
179                throw new MojoExecutionException("Unable to delombok", e);
180            }
181        } else {
182            logger.warn("Skipping " + goal + "; no source to process.");
183        }
184    }
185}
