3   import java.io.File;
4   import java.io.IOException;
5   import java.lang.reflect.InvocationTargetException;
6   import java.nio.charset.Charset;
7   import java.nio.charset.UnsupportedCharsetException;
8   import java.util.ArrayList;
9   import java.util.List;
10  import java.util.Map;
11  
12  import org.apache.commons.lang3.StringUtils;
13  import org.apache.maven.artifact.Artifact;
14  import org.apache.maven.plugin.AbstractMojo;
15  import org.apache.maven.plugin.MojoExecutionException;
16  import org.apache.maven.plugin.logging.Log;
17  import org.apache.maven.plugin.descriptor.PluginDescriptor;
18  import org.apache.maven.plugins.annotations.Component;
19  import org.apache.maven.plugins.annotations.Parameter;
20  import org.apache.maven.project.MavenProject;
21  import org.sonatype.plexus.build.incremental.BuildContext;
22  
23  import lombok.launch.Delombok;
24  
25  /**
26   * Abstract mojo to Delombok java source with lombok annotations.
27   *
28   * @author <a href="mailto:anthony@whitford.com">Anthony Whitford</a>
29   * @see <a href="http://projectlombok.org/features/delombok.html">Delombok</a>
30   */
31  public abstract class AbstractDelombokMojo extends AbstractMojo {
32  
33      /**
34       * Specifies whether the delombok generation should be skipped.
35       */
36      @Parameter(property="lombok.delombok.skip", defaultValue="false", required=true)
37      protected boolean skip;
38  
39      /**
40       * Encoding.
41       */
42      @Parameter(property="lombok.encoding", defaultValue="${project.build.sourceEncoding}", required=true)
43      protected String encoding;
44  
45      /**
46       * Verbose flag.  Print the name of each file as it is being delombok-ed.
47       */
48      @Parameter(property="lombok.verbose", defaultValue="false", required=true)
49      protected boolean verbose;
50   
51      /**
52       * Add output directory flag.  Adds the output directory to the Maven build path.
53       */
54      @Parameter(property="lombok.addOutputDirectory", defaultValue="true", required=true)
55      protected boolean addOutputDirectory;
56  
57      /**
58       * Formatting preferences.
59       */
60      @Parameter
61      protected Map<String, String> formatPreferences;
62  
63      /**
64       * The Maven project to act upon.
65       */
66      @Parameter(property="project", required=true, readonly=true)
67      protected MavenProject project;
68  
69      /**
70       * The plugin dependencies.
71       */
72      @Parameter(property="plugin.artifacts", required=true, readonly=true)
73      private List<Artifact> pluginArtifacts;
74  
75      @Parameter(property="plugin", required=true, readonly=true)
76      protected PluginDescriptor pluginDescriptor;
77  
78      /**
79       * Build Context for improved Maven-Eclipse integration.
80       */
81      @Component
82      private BuildContext buildContext;
83  
84      protected abstract String getGoalDescription ();
85  
86      protected abstract File getOutputDirectory();
87  
88      protected abstract File getSourceDirectory();
89  
90      protected abstract String getSourcePath();
91  
92      protected abstract void addSourceRoot(String path);
93  
94      @Override
95      public void execute() throws MojoExecutionException {
96          final Log logger = getLog();
97          assert null != logger;
98  
99          final String goal = getGoalDescription();
100         logger.debug("Starting " + goal);
101         final File outputDirectory = getOutputDirectory();
102         logger.debug("outputDirectory: " + outputDirectory);
103         final File sourceDirectory = getSourceDirectory();
104         logger.debug("sourceDirectory: " + sourceDirectory);
105         final String sourcePath = getSourcePath();
106         logger.debug("sourcePath: " + sourcePath);
107 
108         if (this.skip) {
109             logger.warn("Skipping " + goal);
110         } else if (sourceDirectory.exists()) {
111             // Build a classPath for delombok...
112             final StringBuilder classPathBuilder = new StringBuilder();
113             for (final Object artifact : project.getArtifacts()) {
114                 classPathBuilder.append(((Artifact)artifact).getFile()).append(File.pathSeparatorChar);
115             }
116             for (final Artifact artifact : pluginArtifacts) {
117                 classPathBuilder.append(artifact.getFile()).append(File.pathSeparatorChar);
118             }
119             // delombok needs tools.jar...
120             final String javaHome = System.getProperty("java.home");
121             final File toolsJar = new File (javaHome,
122                 ".." + File.separatorChar + "lib" + File.separatorChar + "tools.jar");
123             if (toolsJar.exists()) {
124                 try {
125                     pluginDescriptor.getClassRealm().addURL(toolsJar.toURI().toURL());
126                 } catch (final IOException e) {
127                     logger.warn("Unable to add tools.jar; " + toolsJar);
128                 }
129             } else {
130                 logger.warn("Unable to detect tools.jar; java.home is " + javaHome);
131             }
132             final String classPath = classPathBuilder.toString();
133             logger.debug("classpath: " + classPath);
134             try {
135                 final Delombok delombok = new Delombok();
136                 delombok.setVerbose(this.verbose);
137                 delombok.setClasspath(classPath);
138 
139                 if (StringUtils.isNotBlank(this.encoding)) {
140                     try {
141                         delombok.setCharset(this.encoding);
142                     } catch (final UnsupportedCharsetException e) {
143                         logger.error("The encoding parameter is invalid; Please check!", e);
144                         throw new MojoExecutionException("Unknown charset: " + this.encoding, e);
145                     }
146                 } else {
147                     logger.warn("No encoding specified; using default: " + Charset.defaultCharset());
148                 }
149 
150                 if (null != formatPreferences && !formatPreferences.isEmpty()) {
151                     try {
152                         // Construct a list array just like the command-line option...
153                         final List<String> formatOptions = new ArrayList<String>(formatPreferences.size());
154                         for (final Map.Entry<String, String> entry : formatPreferences.entrySet()) {
155                             final String key = entry.getKey();
156                             // "pretty" is an exception -- it has no value...
157                             formatOptions.add( "pretty".equalsIgnoreCase(key) ? key : (key + ':' + entry.getValue()) );
158                         }
159                         delombok.setFormatPreferences(delombok.formatOptionsToMap(formatOptions));
160                     } catch (final Exception e) {
161                         logger.error("The formatPreferences parameter is invalid; Please check!", e);
162                         throw new MojoExecutionException("Invalid formatPreferences: " + this.formatPreferences, e);
163                     }
164                 }
165 
166                 try {
167                     delombok.setOutput(outputDirectory);
168                     delombok.setSourcepath(getSourcePath());
169                     delombok.addDirectory(sourceDirectory);
170                     if (buildContext.hasDelta(sourceDirectory)) {
171                         delombok.delombok();
172                         logger.info(goal + " complete.");
173                     
174                         if (this.addOutputDirectory) {
175                             // adding generated sources to Maven project
176                             addSourceRoot(outputDirectory.getCanonicalPath());
177                             // Notify build context about a file created, updated or deleted...
178                             buildContext.refresh(outputDirectory);
179                         }
180                     } else {
181                         logger.info(goal + " skipped; No deltas detected.");
182                     }
183                 } catch (final IOException e) {
184                     logger.error("Unable to delombok!", e);
185                     throw new MojoExecutionException("I/O problem during delombok", e);
186                 }
187             } catch (final ClassNotFoundException e) {
188                 throw new MojoExecutionException("Unable to delombok", e);
189             } catch (final IllegalAccessException e) {
190                 throw new MojoExecutionException("Unable to delombok", e);
191             } catch (final InvocationTargetException e) {
192                 throw new MojoExecutionException("Unable to delombok", e);
193             } catch (final InstantiationException e) {
194                 throw new MojoExecutionException("Unable to delombok", e);
195             } catch (final NoSuchMethodException e) {
196                 throw new MojoExecutionException("Unable to delombok", e);
197             }
198         } else {
199             logger.warn("Skipping " + goal + "; no source to process.");
200         }
201     }
202 }
